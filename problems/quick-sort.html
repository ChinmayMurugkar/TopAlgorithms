Quick Sort

Divide and Conquer Sorting Algorithm
Problem: Implement Quick Sort algorithm to sort an array of integers.

Algorithm: Quick Sort is a divide-and-conquer algorithm that picks a pivot element and partitions the array around it.

Key Steps:
  1. Choose a pivot element
  2. Partition the array around the pivot
  3. Recursively sort the left and right subarrays

Solution 1 – Basic Quick Sort

Standard implementation with the last element as pivot:

public void quickSort(int[] arr, int low, int high) { if (low < high) { // Find the partition index int pi = partition(arr, low, high); // Recursively sort elements before and after partition quickSort(arr, low, pi - 1); quickSort(arr, pi + 1, high); } } private int partition(int[] arr, int low, int high) { // Choose the rightmost element as pivot int pivot = arr[high]; // Index of smaller element int i = low - 1; for (int j = low; j < high; j++) { // If current element is smaller than or equal to pivot if (arr[j] <= pivot) { i++; // Swap arr[i] and arr[j] int temp = arr[i]; arr[i] = arr[j]; arr[j] = temp; } } // Place pivot in its correct position int temp = arr[i + 1]; arr[i + 1] = arr[high]; arr[high] = temp; return i + 1; } // Wrapper method public void quickSort(int[] arr) { quickSort(arr, 0, arr.length - 1); }
Time Complexity: O(n log n) average, O(n²) worst case
Space Complexity: O(log n) average, O(n) worst case

Solution 2 – Randomized Quick Sort

Using random pivot selection to avoid worst-case scenarios:

public void randomizedQuickSort(int[] arr, int low, int high) { if (low < high) { int pi = randomizedPartition(arr, low, high); randomizedQuickSort(arr, low, pi - 1); randomizedQuickSort(arr, pi + 1, high); } } private int randomizedPartition(int[] arr, int low, int high) { // Choose a random pivot int randomIndex = low + (int)(Math.random() * (high - low + 1)); // Swap random pivot with the last element int temp = arr[randomIndex]; arr[randomIndex] = arr[high]; arr[high] = temp; return partition(arr, low, high); } private int partition(int[] arr, int low, int high) { int pivot = arr[high]; int i = low - 1; for (int j = low; j < high; j++) { if (arr[j] <= pivot) { i++; int temp = arr[i]; arr[i] = arr[j]; arr[j] = temp; } } int temp = arr[i + 1]; arr[i + 1] = arr[high]; arr[high] = temp; return i + 1; }
Time Complexity: O(n log n) average and expected
Space Complexity: O(log n) average

Solution 3 – Three-Way Partition Quick Sort

Efficient for arrays with many duplicate elements:

public void threeWayQuickSort(int[] arr, int low, int high) { if (low < high) { int[] pivotRange = threeWayPartition(arr, low, high); threeWayQuickSort(arr, low, pivotRange[0] - 1); threeWayQuickSort(arr, pivotRange[1] + 1, high); } } private int[] threeWayPartition(int[] arr, int low, int high) { int pivot = arr[low]; int lt = low; // Elements less than pivot int gt = high; // Elements greater than pivot int i = low + 1; // Current element while (i <= gt) { if (arr[i] < pivot) { swap(arr, lt++, i++); } else if (arr[i] > pivot) { swap(arr, i, gt--); } else { i++; } } return new int[]{lt, gt}; } private void swap(int[] arr, int i, int j) { int temp = arr[i]; arr[i] = arr[j]; arr[j] = temp; }
Time Complexity: O(n log n) average
Space Complexity: O(log n) average

Optimization Techniques

1. Median-of-Three Pivot Selection

private int medianOfThree(int[] arr, int low, int high) { int mid = low + (high - low) / 2; // Sort low, mid, high if (arr[low] > arr[mid]) swap(arr, low, mid); if (arr[low] > arr[high]) swap(arr, low, high); if (arr[mid] > arr[high]) swap(arr, mid, high); // Place pivot at high-1 swap(arr, mid, high - 1); return arr[high - 1]; }

2. Insertion Sort for Small Subarrays

public void hybridQuickSort(int[] arr, int low, int high) { if (high - low + 1 <= 10) { insertionSort(arr, low, high); } else if (low < high) { int pi = partition(arr, low, high); hybridQuickSort(arr, low, pi - 1); hybridQuickSort(arr, pi + 1, high); } } private void insertionSort(int[] arr, int low, int high) { for (int i = low + 1; i <= high; i++) { int key = arr[i]; int j = i - 1; while (j >= low && arr[j] > key) { arr[j + 1] = arr[j]; j--; } arr[j + 1] = key; } }
